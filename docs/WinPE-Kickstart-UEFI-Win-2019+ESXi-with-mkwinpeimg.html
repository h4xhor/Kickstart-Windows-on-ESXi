Step 9 - Create BIOS WinPE ISO with Overlay using WinPE Plain boot.wim

Execute the following script: 

                
cd "{ksAttuneBaseDir}/build-{targetServer.fqn}"
    
mkwinpeimg --iso --windows-dir="winpe_plain" \
    --wim="winpe_plain/sources/boot.wim" \
    --overlay="overlay" \
    "winpe.iso"

Step 13 - Extract Plain WinPE boot.img to winpe_staging

Connect via ssh:

ssh {linuxattuneuser}@{attuneosbuildserver}

Execute the following script: 

                
# Extract the boot.img out of the WinPE ISO

cd {ksAttuneBaseDir}/build-{targetServer.fqn}

# Set the filename of the ISO
F="winpe_plain.iso"

# Get the Boot image offset
OFFSET=`isoinfo -d -i $F | grep Bootoff | cut  -c21-`

# Extract the boot.img.
dd if=$F of=boot.img bs=2048 count=8 skip=$OFFSET

Step 14 - WinPE Create BIOS Boot ISO

Execute the following script: 

                
cd {ksAttuneBaseDir}/build-{targetServer.fqn}/winpe_staging

# Create new ISO with answer file
mkisofs -o {ksAttuneBaseDir}/kickstart_{targetServer.fqn}.iso \
    -b boot.img -no-emul-boot -c BOOT.CAT \
    -iso-level 2 -udf -joliet-long -relaxed-filenames -allow-limited-size \
    -J -l -D -N -V "KS_WIN" .
