cd {ksAttuneBaseDir}/build-{kickstartedNode.fqn}

echo "We are in {ksAttuneBaseDir}/build-{kickstartedNode.fqn}."

if [ ! -d "{driversDropDirectory}" ]
then
    echo "{driversDropDirectory} does not exist, skipping."
else
    cd "{driversDropDirectory}"
    NUMBER_FILES=$(ls -1q | wc -l)
    echo "NUMBER_FILES=${NUMBER_FILES}"
    if [ "${NUMBER_FILES}" -eq "0" ]
    then
        echo "no files."
    else
        echo "{driversDropDirectory} exists, copying it's contents to Drivers."
        cp --verbose --recursive {driversDropDirectory}/* Drivers
    fi
fi
